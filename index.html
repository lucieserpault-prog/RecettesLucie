let currentRecipeIngredients = [];

let selectedIng = null;

let editingId = null;

 

// 2. FONCTIONS DE RECHERCHE

function searchIng() {

const query = document.getElementById('ingInput').value.toLowerCase();

const resultsDiv = document.getElementById('searchResults');

resultsDiv.innerHTML = "";

if(query.length < 1) { resultsDiv.style.display = "none"; return; }

 

const filtered = ingredientsDB.filter(i => i.name.toLowerCase().includes(query));

filtered.forEach(i => {

const div = document.createElement('div');

div.className = "search-item";

div.innerText = i.name;

div.onclick = () => selectIngredient(i);

resultsDiv.appendChild(div);

});

resultsDiv.style.display = "block";

}

 

function selectIngredient(ing) {

selectedIng = ing;

document.getElementById('selectedIngName').innerText = "Slectionn : " + ing.name;

document.getElementById('searchResults').style.display = "none";

document.getElementById('ingInput').value = "";

}

 

// 3. GESTION DES INGRDIENTS DANS LA RECETTE

function addIngredientToList() {

const qty = parseFloat(document.getElementById('ingQty').value);

if(!selectedIng || !qty) return;

 

const factor = qty / selectedIng.base;

const newIng = {

...selectedIng,

actualQty: qty,

calcCal: selectedIng.cal * factor,

calcProt: selectedIng.prot * factor,

calcGlu: selectedIng.glu * factor,

calcLip: selectedIng.lip * factor

};

 

currentRecipeIngredients.push(newIng);

renderCurrentIngredients();

updateLiveMacros();

// Reset

selectedIng = null;

document.getElementById('selectedIngName').innerText = "";

document.getElementById('ingQty').value = "";

}

 

function renderCurrentIngredients() {

const list = document.getElementById('addedIngredientsList');

list.innerHTML = currentRecipeIngredients.map((ing, idx) => `

<div class="ing-row">

<span>${ing.name} (${ing.actualQty}g/u)</span>

<span class="remove-btn" onclick="removeIng(${idx})">✖</span>

</div>

`).join('');

}

 

function removeIng(idx) {

currentRecipeIngredients.splice(idx, 1);

renderCurrentIngredients();

updateLiveMacros();

}

 

function updateLiveMacros() {

let tCal = 0, tProt = 0, tGlu = 0, tLip = 0;

const servings = parseFloat(document.getElementById('rServings').value) || 1;

 

currentRecipeIngredients.forEach(i => {

tCal += i.calcCal; tProt += i.calcProt; tGlu += i.calcGlu; tLip += i.calcLip;

});

 

document.getElementById('liveCal').innerText = Math.round(tCal / servings);

document.getElementById('liveProt').innerText = Math.round(tProt / servings);

document.getElementById('liveGlu').innerText = Math.round(tGlu / servings);

document.getElementById('liveLip').innerText = Math.round(tLip / servings);

}

 

// 4. SAUVEGARDE ET CARNET

async function saveRecipe() {

const name = document.getElementById('rName').value;

if(!name || currentRecipeIngredients.length === 0) { alert("Nom ou ingrdients manquants !"); return; }

 

const imgFile = document.getElementById('rImage').files[0];

let imgData = "";

if(imgFile) {

imgData = await toBase64(imgFile);

} else if (editingId) {

const old = getRecipes().find(r => r.id === editingId);

imgData = old.image;

}

 

const recipe = {

id: editingId || Date.now(),

name,

theme: document.getElementById('rTheme').value,

subTheme: document.getElementById('rSubTheme').value,

servings: document.getElementById('rServings').value,

ingredients: [...currentRecipeIngredients],

instructions: document.getElementById('rInstructions').value,

image: imgData,

macros: {

cal: document.getElementById('liveCal').innerText,

prot: document.getElementById('liveProt').innerText,

glu: document.getElementById('liveGlu').innerText,

lip: document.getElementById('liveLip').innerText

}

};

 

let recipes = getRecipes();

if(editingId) {

recipes = recipes.map(r => r.id === editingId ? recipe : r);

} else {

recipes.push(recipe);

}

 

localStorage.setItem('chefRecipes', JSON.stringify(recipes));

resetForm();

renderCarnet();

}

 

function renderCarnet() {

const carnet = document.getElementById('carnetList');

const recipes = getRecipes();

carnet.innerHTML = recipes.map(r => `

<div class="recipe-card">

${r.image ? `<img src="${r.image}" class="recipe-img">` : '<div class="recipe-img" style="display:flex;align-items:center;justify-content:center">Pas de photo</div>'}

<div class="recipe-content">

<span class="tag">${r.theme}</span><span class="tag">${r.subTheme}</span>

<h3>${r.name}</h3>

<p><strong>${r.macros.cal} kcal</strong> / part (${r.servings} parts)</p>

<div class="recipe-actions">

<button class="btn btn-save btn-action" onclick="loadEdit(${r.id})">Modifier</button>

<button class="btn btn-add btn-action" style="background:#3498db" onclick="generateShopping(${r.id})">Courses</button>

<button class="btn btn-action" style="background:#e74c3c; color:white" onclick="deleteRecipe(${r.id})">Supprimer</button>

</div>

</div>

</div>

`).join('');

}

 

function generateShopping(id) {

const r = getRecipes().find(r => r.id === id);

const modal = document.getElementById('shoppingModal');

const content = document.getElementById('shoppingContent');

content.innerHTML = r.ingredients.map(i => `<div>☐ ${i.name} : ${i.actualQty}</div>`).join('');

modal.classList.remove('hidden');

}

 

function loadEdit(id) {

const r = getRecipes().find(r => r.id === id);

editingId = id;

document.getElementById('rName').value = r.name;

document.getElementById('rTheme').value = r.theme;

document.getElementById('rSubTheme').value = r.subTheme;

document.getElementById('rServings').value = r.servings;

document.getElementById('rInstructions').value = r.instructions;

currentRecipeIngredients = [...r.ingredients];

document.getElementById('formTitle').innerText = "Modifier la recette";

document.getElementById('btnCancelEdit').classList.remove('hidden');

renderCurrentIngredients();

updateLiveMacros();

window.scrollTo(0,0);

}

 

function resetForm() {

editingId = null;

document.getElementById('rName').value = "";

document.getElementById('rInstructions').value = "";

document.getElementById('rImage').value = "";

currentRecipeIngredients = [];

document.getElementById('formTitle').innerText = "Nouvelle Cration";

document.getElementById('btnCancelEdit').classList.add('hidden');

renderCurrentIngredients();

updateLiveMacros();

}

 

function deleteRecipe(id) {

if(confirm("Supprimer cette recette ?")) {

let recipes = getRecipes().filter(r => r.id !== id);

localStorage.setItem('chefRecipes', JSON.stringify(recipes));

renderCarnet();

}

}

 

function getRecipes() { return JSON.parse(localStorage.getItem('chefRecipes')) || []; }

function toBase64(file) {

return new Promise((resolve, reject) => {

const reader = new FileReader();

reader.readAsDataURL(file);

reader.onload = () => resolve(reader.result);

reader.onerror = error => reject(error);

});

}

 

function exportData() {

const data = localStorage.getItem('chefRecipes');

const blob = new Blob([data], {type: 'text/plain'});

const a = document.createElement('a');

a.href = URL.createObjectURL(blob);

a.download = 'mon_carnet_backup.txt';

a.click();

}
